142  cell *tmp;
143  if(!op) return EOF;
144  if(l && depth > MAX_RECURSION_DEPTH) {
145  lisp_printf(l, o, depth, "%r<PRINT-DEPTH-EXCEEDED:%d>%t", (intptr_t) depth);
146  return -1;
147  }
148  switch(op->type) {
149  case INTEGER: lisp_printf(l, o, depth, "%m%d", get_int(op)); break;
150  case FLOAT: lisp_printf(l, o, depth, "%m%f", get_float(op)); break;
151  case CONS: if(depth && o->pretty) lisp_printf(l, o, depth, "\n%* ");
152  io_putc('(', o);
153  for(;;) {
154  printer(l, o, car(op), depth + 1);

156  io_putc(')', o);

158  }
159  op = cdr(op);
160  if(!is_cons(op)) {
161  lisp_printf(l, o, depth, " . %S)", op);
162  break;
163  }
164  io_putc(' ', o);
165  }
166  break;
167  case SYMBOL: if(is_nil(op)) lisp_printf(l, o, depth, "%rnil");
168  else lisp_printf(l, o, depth, "%y%s", get_sym(op));
169  break;
170  case STRING: print_escaped_string(l, o, depth, get_str(op)); break;
171  case SUBR: lisp_printf(l, o, depth, "%B<SUBR:%d>", get_int(op)); break;
172  case PROC: case FPROC:
173  lisp_printf(l, o, depth+1,
174  is_proc(op) ? "(%ylambda%t %S %S " :
175  "(%yflambda%t %S %S ",
176  get_func_docstring(op), get_proc_args(op));
177  for(tmp = get_proc_code(op); !is_nil(tmp); tmp = cdr(tmp)) {
178  printer(l, o, car(tmp), depth+1);
179  if(!is_nil(cdr(tmp)))
180  io_putc(' ', o);
181  }
182  io_putc(')', o);
183  break;
184  case HASH: lisp_printf(l, o, depth, "%H", get_hash(op)); break;
185  case IO: lisp_printf(l, o, depth, "%B<IO:%s:%d>",
186  op->close? "CLOSED" :
187  (is_in(op)? "IN": "OUT"), get_int(op)); break;
188  case USERDEF: if(l && l->ufuncs[get_user_type(op)].print)
189  (l->ufuncs[get_user_type(op)].print)(o, depth, op);
190  else lisp_printf(l, o, depth, "<USER:%d:%d>",
191  get_user_type(op), get_int(op)); break;
192  case INVALID:
193  default: FATAL("internal inconsistency");
194  }
195  return lisp_printf(l, o, depth, "%t") == EOF ? EOF : 0;
196 }
